proj_B <- function(R, v){
  R_hat_minus = R - tcrossprod(v)
  eigen_R = eigen(R_hat_minus)
  
  V <- eigen_R$vectors
  D_plus <- diag(pmax(eigen_R$values, 0))
  
  R_hat_plus <- V %*% D_plus %*% t(V) + tcrossprod(v)
  return(R_hat_plus)
}

  (R + t(R)) / 2
